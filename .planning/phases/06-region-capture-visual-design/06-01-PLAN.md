lib/pin-position.ts (pure, DOM-free, node:test-safe — NO top-level window/document/chrome; the caller in Plan 04's pin.ts supplies the live rect + scroll offsets as parameters):
  export function matchesUrlPath(noteUrl:string, pageUrl:string): boolean;
  export function computePinPosition(
    anchorRect: { x:number; y:number; width:number; height:number } | null,
    storedRect: { x:number; y:number; width:number; height:number } | null,
    scrollX: number,
    scrollY: number,
    orphaned: boolean
  ): { left:number; top:number; orphaned:boolean };
  // anchorRect non-null + !orphaned → element-anchored (live getBoundingClientRect, position:fixed coords → {left:anchorRect.x, top:anchorRect.y})
  // anchorRect null (selector matched nothing) → orphaned:true, position from storedRect minus scroll (last-known page-absolute rect → fixed coords): {left: storedRect.x - scrollX, top: storedRect.y - scrollY}
  // free note (orphaned false, anchorRect carries the free viewportCoords as x/y, width/height 0) → {left:anchorRect.x, top:anchorRect.y}
  // NEVER access window.scrollX inside the function — caller passes scrollX/scrollY
